5. Formatting Code
-------------------

SELECT D.DEPARTMENT_NAME, E.EMPLOYEE_ID, E.FIRST_NAME, 
E.LAST_NAME, E.EMAIL, E.HIRE_DATE, E.JOB_ID, E.SALARY, 
E.MANAGER_ID, L.LOCATION_ID, C.COUNTRY_ID, C.COUNTRY_NAME, 
L.STREET_ADDRESS, L.POSTAL_CODE, L.CITY, L.STATE_PROVINCE, L.COUNTRY_ID
FROM DEPARTMENTS D, LOCATIONS L, EMPLOYEES E, COUNTRIES C 
WHERE L.LOCATION_ID = D.LOCATION_ID AND E.EMPLOYEE_ID = 
D.MANAGER_ID AND D.DEPARTMENT_ID = E.DEPARTMENT_ID AND 
C.COUNTRY_ID  = L.COUNTRY_ID;

For HR:

DECLARE
EMP_NAME VARCHAR2(10);
CURSOR C1 IS SELECT LAST_NAME FROM EMPLOYEES
WHERE DEPARTMENT_ID = 20;
BEGIN
OPEN C1;
LOOP
FETCH C1 INTO EMP_NAME;
EXIT WHEN C1%NOTFOUND;
HTP.PRINT(EMP_NAME);
END LOOP;
END;

SELECT
  "HR"."EMPLOYEES"."LAST_NAME",
  "HR"."DEPARTMENTS"."DEPARTMENT_NAME",
  "HR". "LOCATIONS"."CITY",
  "HR"."COUNTRIES"."COUNTRY_NAME"
FROM
  "HR"."EMPLOYEES",
  "HR"."DEPARTMENTS",
  "HR"."LOCATIONS",
  "HR"."COUNTRIES"
WHERE ("HR"."EMPLOYEES"."DEPARTMENT_ID" = "HR"."DEPARTMENTS"."DEPARTMENT_ID")
AND ("HR"."DEPARTMENTS"."MANAGER_ID"    = "HR"."EMPLOYEES"."EMPLOYEE_ID")
AND ("HR"."DEPARTMENTS"."LOCATION_ID"   = "HR"."LOCATIONS"."LOCATION_ID")
AND ("HR"."LOCATIONS"."COUNTRY_ID"      = "HR"."COUNTRIES"."COUNTRY_ID")
AND (("HR"."COUNTRIES"."COUNTRY_NAME"  = 'United Kingdom')
OR (("HR"."LOCATIONS"."CITY"           = 'Seattle')
OR ("HR"."LOCATIONS"."CITY" LIKE '%San Francisco%')))
